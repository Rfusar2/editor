    git log -2                         return last two commits
    git log --grep="example"           for searching into a comment area of the commit
    git log --pretty-format="%H"       for filtering the return value of log action
            %H	                       Hash completo del commit
            %h	                       Hash abbreviato
            %T	                       Hash completo dell’albero (tree)
            %t	                       Hash abbreviato dell’albero
            %P	                       Hash completi dei genitori
            %p	                       Hash abbreviati dei genitori
            %an	                       Nome autore
            %ae	                       Email autore

            %ad	                       Data autore (formato locale)
            %ar	                       Data autore relativa (es. “2 days ago”)
            %cn	                       Nome committer
            %ce	                       Email committer
            %cd	                       Data committer
            %cr	                       Data committer relativa
            %s	                       Messaggio del commit (subject)
            %b	                       Body del commit (messaggio completo)
            %N	                       Note del commit
            %D	                       Riferimenti (tag, branch)

[clear commits]
git show <commit-hash>
git diff <commit1:vecchio> <commit2:nuovo>
git rebase -i HEAD~3       (last 3 commits)
git rebase -i --root       (all branches) open the terminal with the possibility of:

    fixup                                 join commits with no comments
    squash                                join commits with comments
    pickup                                stay into current branch

[temp saves without commits]
git stash                                 save the canches temporary
git stash list                            get the list of the stashes
git stash applay stash@{1}                example for applaying
git stash drop stash@{1}                  example for deleting
git stash clear                           delete all stashes
